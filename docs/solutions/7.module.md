##### exercise 3 #####
> Caution! 
```javascript
['1','5','7'].map(parseInt);  //[10, NaN, NaN] WRONG!

//longer solution:
function returnNumber(value) {
  return parseInt(value,10);  
}

var result = ['1','5','7'].map(returnNumber);
console.log(result);

//shorter solution:
var result = ['1','5','7'].map(Number); //uses Class Constructor to convert Types
console.log(result);

```
