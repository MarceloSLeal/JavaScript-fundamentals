```javascript
//Ex5:
var str = '0,1,2,3,4,5';
var len = str.length;
for (var i=0; i < len; i++) {
  var value = str.charAt(i);
  if (isNaN(value)) {
    console.log(value + ' is not a number');
  } else {
    console.log(value + ' is ' + (value % 2 ? 'odd' : 'even'));
  }
}
```
